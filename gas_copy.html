function doGet(e){
  var action=(e&&e.parameter&&e.parameter.action)||'getAll';
  var callback=(e&&e.parameter&&e.parameter.callback)||'';
  var result;
  try{if(action==='getAll'){result=getAllData();}else{result={ok:true,msg:'ok'};}}
  catch(err){result={ok:false,error:err.message};}
  var json=JSON.stringify(result);
  if(callback)return ContentService.createTextOutput(callback+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT);
  return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  var body={};
  try{body=JSON.parse(e.postData.contents);}catch(err){}
  if(body.events&&body.events.length>0){
    body.events.forEach(function(event){
      if(event.type==='message'&&event.source&&event.source.userId){
        var uid=event.source.userId;
        var dname='';
        try{
          var props=PropertiesService.getScriptProperties();
          var tok=props.getProperty('LINE_TOKEN');
          if(tok){
            var r=UrlFetchApp.fetch('https://api.line.me/v2/bot/profile/'+uid,{headers:{'Authorization':'Bearer '+tok},muteHttpExceptions:true});
            if(r.getResponseCode()===200){var pr=JSON.parse(r.getContentText());dname=pr.displayName||'';}
          }
        }catch(err){}
        saveLineUserId(uid,dname,event.message?event.message.text:'');
      }
    });
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }
  var action=body.action||'',result;
  try{
    if(action==='getAll'){result=getAllData();}
    else if(action==='saveBookings'){saveSheet('\u4e88\u7d04\u8868',JSON.parse(body.rows));result={ok:true};}
    else if(action==='saveCustomers'){saveSheet('\u60a3\u8005',JSON.parse(body.rows));result={ok:true};}
    else if(action==='saveUriage'){saveSheet('\u58f2\u4e0a',JSON.parse(body.rows));result={ok:true};}
    else if(action==='resetBookings'){resetBookings();result={ok:true};}
    else if(action==='lineNotify'){result=sendLineNotify(body.token,body.message);}
    else if(action==='lineNotifyV2'){result=sendLineMessagingAPI(body.token,body.userId,body.message);}
    else if(action==='saveLineSettings'){saveLineSettings();result={ok:true};}
    else if(action==='getLineUsers'){result=getLineUsers();}
    else{result={ok:false,error:'unknown action'};}
  }catch(err){result={ok:false,error:err.message};}
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

  if(!sheet){

  }
  var data=sheet.getDataRange().getValues();
  var now=new Date();
  for(var i=1;i<data.length;i++){
    if(data[i][0]===userId){

function saveSheet(name,rows){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var s=ss.getSheetByName(name);
  if(!s)s=ss.insertSheet(name);
  s.clearContents();
  if(!rows||rows.length===0)return;
  var san=rows.map(function(row){return row.map(function(cell){
    if(cell===null||cell===undefined)return'';
    if(cell instanceof Date){var y=cell.getFullYear(),mo=String(cell.getMonth()+1).padStart(2,'0'),d=String(cell.getDate()).padStart(2,'0');return y+'-'+mo+'-'+d;}
    return String(cell);
  });});
  var range=s.getRange(1,1,san.length,san[0].length);
  var fmt=san.map(function(r){return r.map(function(){return'@';});});
  range.setNumberFormats(fmt);
  range.setValues(san);
}

function dailyLineAlert(){
  var props=PropertiesService.getScriptProperties();
  var token=props.getProperty('LINE_TOKEN');
  var userId=props.getProperty('LINE_USER_ID');
  if(!token||!userId){Logger.log('\u30c8\u30fc\u30af\u30f3\u672a\u8a2d\u5b9a');return;}
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var today=new Date();today.setHours(0,0,0,0);
  var alerts=[];
  var bSheet=ss.getSheetByName('\u4e88\u7d04\u8868');
  if(bSheet){
    var bData=bSheet.getDataRange().getValues();
    var bHdr=bData[0];
    var dateIdx=bHdr.indexOf('\u65e5\u4ed8'),nameIdx=bHdr.indexOf('\u60a3\u8005\u540d'),idIdx=bHdr.indexOf('\u8a3a\u5bdf\u5238No'),kubunIdx=bHdr.indexOf('\u533a\u5206');
    if(dateIdx>-1&&nameIdx>-1){
      var lastVisit={};
      bData.slice(1).forEach(function(row){
        var kubun=String(row[kubunIdx]||'');
        if(kubun.indexOf('\u7d99\u7d9a')>-1||kubun.indexOf('\u30ad\u30e3\u30f3\u30bb\u30eb')>-1)return;
        var dateVal=row[dateIdx];
        if(!dateVal)return;
        var d=dateVal instanceof Date?new Date(dateVal):new Date(String(dateVal));
        if(isNaN(d.getTime()))return;
        d.setHours(0,0,0,0);
        if(d>today)return;
        var key=String(row[idIdx]||'')||String(row[nameIdx]||'');
        if(!lastVisit[key]||d>lastVisit[key].date)lastVisit[key]={date:d,name:String(row[nameIdx]||''),id:String(row[idIdx]||'')};
      });
      Object.values(lastVisit).forEach(function(v){
        var diff=Math.floor((today-v.date)/(1000*60*60*24));
        if(diff>=20&&diff<=21){
          var ds=Utilities.formatDate(v.date,'Asia/Tokyo','M/d');
          alerts.push('\u30fb'+v.name+'\uff08'+v.id+'\u53f7\uff09\u524d\u56de:'+ds+' '+diff+'\u65e5\u7d4c\u904e');
        }
      });
    }
  }
  if(alerts.length>0){
    var msg='\u300c\u5009\u6cbb\u6574\u9aa8\u9662\u300d\n\u26a0\ufe0f \u660e\u65e5\u3067\u521d\u8a3a\u6599\u767a\u751f\u30a2\u30e9\u30fc\u30c8\n'+Utilities.formatDate(today,'Asia/Tokyo','M\u6708d\u65e5(E)')+'\n\n'+alerts.join('\n')+'\n\n\u2192 \u6765\u9662\u3092\u4fc3\u3057\u3066\u304f\u3060\u3055\u3044\uff01';
    sendLineMessagingAPI(token,userId,msg);
  }else{

  }
}
function sendDayBeforeReminders(){
  var props=PropertiesService.getScriptProperties();
  var token=props.getProperty('LINE_TOKEN');
  if(!token){Logger.log('\u30c8\u30fc\u30af\u30f3\u672a\u8a2d\u5b9a');return;}
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var bSheet=ss.getSheetByName('\u4e88\u7d04\u8868');
  if(!bSheet){Logger.log('\u4e88\u7d04\u8868\u30b7\u30fc\u30c8\u306a\u3057');return;}
  var today=new Date();today.setHours(0,0,0,0);
  var tomorrow=new Date(today);tomorrow.setDate(tomorrow.getDate()+1);
  var tomorrowStr=Utilities.formatDate(tomorrow,'Asia/Tokyo','yyyy-MM-dd');
  var tomorrowDisp=Utilities.formatDate(tomorrow,'Asia/Tokyo','M\u6708d\u65e5(E)');
  var lineUsers={};
  var lSheet=ss.getSheetByName('LINE_IDs');
  if(lSheet){
    var lData=lSheet.getDataRange().getValues();
    lData.slice(1).forEach(function(row){if(row[0]&&row[1])lineUsers[String(row[1]).trim()]=String(row[0]);});
  }
  var data=bSheet.getDataRange().getValues();
  var hdr=data[0];
  var dateIdx=hdr.indexOf('\u65e5\u4ed8'),timeIdx=hdr.indexOf('\u6642\u9593');
  var nameIdx=hdr.indexOf('\u60a3\u8005\u540d'),kubunIdx=hdr.indexOf('\u533a\u5206');
  if(dateIdx<0){Logger.log('\u30d8\u30c3\u30c0\u30fc\u672a\u767a\u898b');return;}
  var byPatient={},seen={};
  data.slice(1).forEach(function(row){
    var dateVal=String(row[dateIdx]||'').trim();
    var kubun=String(row[kubunIdx]||'');
    if(kubun.indexOf('\u7d99\u7d9a')>-1||kubun.indexOf('\u30ad\u30e3\u30f3\u30bb\u30eb')>-1)return;
    if(dateVal!==tomorrowStr)return;
    var name=String(row[nameIdx]||'').trim();
    var time=String(row[timeIdx]||'').trim();
    if(!name||!time||seen[name+'_'+time])return;
    seen[name+'_'+time]=true;
    if(!byPatient[name])byPatient[name]=[];
    byPatient[name].push(time);
  });
  if(!Object.keys(byPatient).length){Logger.log('\u660e\u65e5('+tomorrowStr+')\u306e\u4e88\u7d04\u306a\u3057');return;}
  var sent=0,skipped=[];
  Object.keys(byPatient).forEach(function(name){
    var times=byPatient[name];
    var targetId=lineUsers[name]||null;
    if(!targetId){
      var lastName=name.split(/[\u3000 ]/)[0];
      var found=Object.keys(lineUsers).find(function(k){return k.replace(/[\u3000 ]/,'').indexOf(lastName)===0;});
      if(found)targetId=lineUsers[found];
    }
    if(!targetId){skipped.push(name);return;}
    var msg='\u300c\u5009\u6cbb\u6574\u9aa8\u9662\u300d\n'+tomorrowDisp+'\u306e\u3054\u4e88\u7d04\u30ea\u30de\u30a4\u30f3\u30c9\u3067\u3059\ud83d\ude0a\n\n\u23f0 '+times.join(' / ')+'\n\n\u304a\u6c17\u3092\u3064\u3051\u3066\u304a\u8d8a\u3057\u304f\u3060\u3055\u3044\u3002\n\u203b\u81ea\u52d5\u9001\u4fe1\u306e\u305f\u3081\u8fd4\u4fe1\u4e0d\u8981\u3067\u3059\u3002';
    var result=sendLineMessagingAPI(token,targetId,msg);
    if(result.ok){sent++;}else{skipped.push(name+'(\u9001\u4fe1\u30a8\u30e9\u30fc)');}
  });
  Logger.log('\u30ea\u30de\u30a4\u30f3\u30c9\u5b8c\u4e86 \u9001\u4fe1:'+sent+'\u4ef6 \u672a\u767b\u9332:'+skipped.length+'\u4ef6');
  var ownerUserId=props.getProperty('LINE_USER_ID');
  if(ownerUserId){
    var summary='\u300c\u5009\u6cbb\u6574\u9aa8\u9662\u300d\n\ud83d\udcca \u524d\u65e5\u30ea\u30de\u30a4\u30f3\u30c9\u5b8c\u4e86\n'+tomorrowDisp+'\n\u9001\u4fe1:'+sent+'\u4ef6';
    if(skipped.length>0)summary+='\nLINE\u672a\u767b\u9332:\n\u30fb'+skipped.join('\n\u30fb');
    sendLineMessagingAPI(token,ownerUserId,summary);
  }
}

function sendLineMessagingAPI(token,userId,message){
  if(!token||!userId||!message)return{ok:false,error:'missing params'};
  try{
    var payload=JSON.stringify({to:userId,messages:[{type:'text',text:message}]});
    var res=UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push',{
      method:'post',
      headers:{'Content-Type':'application/json','Authorization':'Bearer '+token},
      payload:payload
    });
    return{ok:res.getResponseCode()===200,status:res.getResponseCode()};
  }catch(err){return{ok:false,error:err.message};}
}
function sendLineNotify(token,message){
  if(!token||!message)return{ok:false,error:'missing'};
  try{
    var res=UrlFetchApp.fetch('https://notify-api.line.me/api/notify',{method:'post',headers:{'Authorization':'Bearer '+token},payload:{message:message}});
    return{ok:res.getResponseCode()===200};
  }catch(err){return{ok:false,error:err.message};}
}
