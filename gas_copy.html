function doGet(e){
  var action=(e&&e.parameter&&e.parameter.action)||'getAll';
  var callback=(e&&e.parameter&&e.parameter.callback)||'';
  var result;
  try{if(action==='getAll'){result=getAllData();}else{result={ok:true,msg:'ok'};}}
  catch(err){result={ok:false,error:err.message};}
  var json=JSON.stringify(result);
  if(callback)return ContentService.createTextOutput(callback+'('+json+')').setMimeType(ContentService.MimeType.JAVASCRIPT);
  return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  var body={};try{body=JSON.parse(e.postData.contents);}catch(err){}
  var action=body.action||'',result;
  try{
    if(action==='getAll'){result=getAllData();}
    else if(action==='saveBookings'){saveSheet('\u4e88\u7d04\u8868',JSON.parse(body.rows));result={ok:true};}

    else if(action==='lineNotifyV2'){result=sendLineMessagingAPI(body.token,body.userId,body.message);}

}

function saveSheet(name,rows){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var s=ss.getSheetByName(name);
  if(!s)s=ss.insertSheet(name);
  s.clearContents();
  if(!rows||rows.length===0)return;
  var san=rows.map(function(row){return row.map(function(cell){
    if(cell===null||cell===undefined)return'';
    if(cell instanceof Date){var y=cell.getFullYear(),mo=String(cell.getMonth()+1).padStart(2,'0'),d=String(cell.getDate()).padStart(2,'0');return y+'-'+mo+'-'+d;}
    return String(cell);
  });});
  var range=s.getRange(1,1,san.length,san[0].length);
  var fmt=san.map(function(r){return r.map(function(){return'@';});});
  range.setNumberFormats(fmt);
  range.setValues(san);
}

function dailyLineAlert(){
  var props=PropertiesService.getScriptProperties();
  var token=props.getProperty('LINE_TOKEN');
  var userId=props.getProperty('LINE_USER_ID');
  if(!token||!userId){Logger.log('トークン未設定 - saveLineSettings()を実行してください');return;}

  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var today=new Date();
  today.setHours(0,0,0,0);
  var alerts=[];

  // 予約表から各患者の最終来院日を計算
  var bSheet=ss.getSheetByName('予約表');
  if(bSheet){
    var bData=bSheet.getDataRange().getValues();
    var bHdr=bData[0];
    var dateIdx=bHdr.indexOf('日付');
    var nameIdx=bHdr.indexOf('患者名');
    var idIdx=bHdr.indexOf('診察券No');
    var kubunIdx=bHdr.indexOf('区分');
    if(dateIdx>-1&&nameIdx>-1){
      // 患者ごとの最終来院日を集計
      var lastVisit={};
      bData.slice(1).forEach(function(row){
        var kubun=String(row[kubunIdx]||'');
        if(kubun.indexOf('継続')>-1||kubun.indexOf('キャンセル')>-1)return;
        var dateVal=row[dateIdx];
        if(!dateVal)return;
        var d;
        if(dateVal instanceof Date){d=new Date(dateVal);}
        else{d=new Date(String(dateVal));}
        if(isNaN(d.getTime()))return;
        d.setHours(0,0,0,0);
        // 未来の予約は除外
        if(d>today)return;
        var key=String(row[idIdx]||'')||String(row[nameIdx]||'');
        if(!lastVisit[key]||d>lastVisit[key].date){
          lastVisit[key]={date:d,name:String(row[nameIdx]||''),id:String(row[idIdx]||'')};
        }
      });
      // 20日チェック
      Object.values(lastVisit).forEach(function(v){
        var diff=Math.floor((today-v.date)/(1000*60*60*24));
        if(diff>=20&&diff<=21){
          var dateStr=Utilities.formatDate(v.date,'Asia/Tokyo','M/d');
          alerts.push('・'+v.name+'（'+v.id+'号）前回:'+dateStr+' '+diff+'日経過');
        }
      });
    }
  }

  if(alerts.length>0){
    var msg='【倉治整骨院】\n⚠️ 明日で初診料発生アラート\n'+Utilities.formatDate(today,'Asia/Tokyo','M月d日(E)')+'\n\n'+alerts.join('\n')+'\n\n→ 来院を促してください！';
    sendLineMessagingAPI(token,userId,msg);
    Logger.log('アラート送信：'+alerts.length+'件');
  }else{
    Logger.log('20日アラート対象なし（本日 '+Utilities.formatDate(today,'Asia/Tokyo','M/d')+')');
  }
}

function sendDayBeforeReminders(){
  var props=PropertiesService.getScriptProperties();
  var token=props.getProperty('LINE_TOKEN');
  var myUserId=props.getProperty('LINE_USER_ID');
  if(!token){Logger.log('トークン未設定');return;}

  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var today=new Date();today.setHours(0,0,0,0);
  var tomorrow=new Date(today);tomorrow.setDate(tomorrow.getDate()+1);
  var tomorrowStr=Utilities.formatDate(tomorrow,'Asia/Tokyo','yyyy-MM-dd');
  var tomorrowDisp=Utilities.formatDate(tomorrow,'Asia/Tokyo','M月d日(E)');

  // 予約表を取得
  var bSheet=ss.getSheetByName('予約表');
  if(!bSheet){Logger.log('予約表シートなし');return;}
  var bData=bSheet.getDataRange().getValues();
  var bHdr=bData[0];
  var dateIdx=bHdr.indexOf('日付'),timeIdx=bHdr.indexOf('時間');
  var nameIdx=bHdr.indexOf('患者名'),idIdx=bHdr.indexOf('診察券No'),kubunIdx=bHdr.indexOf('区分');

  // 患者シートからLINEユーザーIDを取得
  var lineUids={};
  var pSheet=ss.getSheetByName('患者');
  if(pSheet){
    var pData=pSheet.getDataRange().getValues();
    var pHdr=pData[0];
    var pIdIdx=pHdr.indexOf('診察券No'),pLineUidIdx=pHdr.indexOf('LINEユーザーID');
    if(pIdIdx>-1&&pLineUidIdx>-1){
      pData.slice(1).forEach(function(row){
        var pid=String(row[pIdIdx]||'').trim();
        var uid=String(row[pLineUidIdx]||'').trim();
        if(pid&&uid&&uid.startsWith('U'))lineUids[pid]=uid;
      });
    }
  }

  // 明日の予約を収集
  var appointments=[];var seen={};
  bData.slice(1).forEach(function(row){
    var dateVal=String(row[dateIdx]||'').trim();
    var kubun=String(row[kubunIdx]||'');
    if(kubun.indexOf('継続')>-1||kubun.indexOf('キャンセル')>-1)return;
    if(dateVal!==tomorrowStr)return;
    var name=String(row[nameIdx]||'').trim();
    var time=String(row[timeIdx]||'').trim();
    var pid=String(row[idIdx]||'').trim();
    if(!name||!time)return;
    var key=name+'_'+time;
    if(seen[key])return;seen[key]=true;
    appointments.push({name:name,time:time,pid:pid});
  });

  if(!appointments.length){Logger.log('明日の予約なし');return;}

  var sentCount=0,ownCount=0;
  // 患者個別に送信
  appointments.forEach(function(a){
    var uid=lineUids[a.pid]||'';
    if(uid){
      var msg='【倉治整骨院】\n'+a.name+'様\n明日('+tomorrowDisp+') '+a.time+'〜のご予約を\nお待ちしております😊\n\n気をつけてお越しください🙏\n※自動送信のため返信不要です。';
      var result=sendLineMessagingAPI(token,uid,msg);
      if(result.ok)sentCount++;
      else Logger.log('送信失敗: '+a.name+' '+JSON.stringify(result));
    }else{
      ownCount++;
    }
  });

  // LINEユーザーID未登録の患者はまとめて自分のLINEに通知
  if(ownCount>0&&myUserId){
    var unregistered=appointments.filter(function(a){return !lineUids[a.pid];});
    var msg2='【倉治整骨院】\n📅 明日('+tomorrowDisp+')のLINE未登録患者\n\n';
    unregistered.forEach(function(a){msg2+='・'+a.time+' '+a.name+'様\n';});
    msg2+='\n→ LINEユーザーIDを登録すると自動でリマインドが届きます';
    sendLineMessagingAPI(token,myUserId,msg2);
  }

  Logger.log('患者個別送信:'+sentCount+'件 / 未登録:'+ownCount+'件');
}

// 両方のトリガーをまとめて設定
function setupAllTriggers(){
  // 既存トリガーを全削除
  ScriptApp.getProjectTriggers().forEach(function(t){ScriptApp.deleteTrigger(t);});
  // 毎朝8時：20日アラート
  ScriptApp.newTrigger('dailyLineAlert').timeBased().everyDays(1).atHour(8).create();
  // 毎日18時：前日リマインド
  ScriptApp.newTrigger('sendDayBeforeReminders').timeBased().everyDays(1).atHour(18).create();
  Logger.log('✅ トリガー設定完了：毎朝8時(20日アラート) + 毎日18時(前日リマインド)');
}

function setupDailyTrigger(){
  // 既存のdailyLineAlertトリガーを削除

    .everyDays(1)
    .atHour(8)
    .create();
  Logger.log('✅ 毎朝8時の自動アラートトリガーを設定しました！');
}

function sendLineMessagingAPI(token,userId,message){
  if(!token||!userId||!message)return{ok:false,error:'missing params'};
  try{
    var payload=JSON.stringify({to:userId,messages:[{type:'text',text:message}]});
    var res=UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push',{
      method:'post',
      headers:{'Content-Type':'application/json','Authorization':'Bearer '+token},
      payload:payload
    });
    var code=res.getResponseCode();
    return{ok:code===200,status:code};
  }catch(err){
    return{ok:false,error:err.message};
  }
}
function sendLineNotify(token,message){
  if(!token||!message)return{ok:false,error:'token or message missing'};
  try{
    var res=UrlFetchApp.fetch('https://notify-api.line.me/api/notify',{
      method:'post',
      headers:{'Authorization':'Bearer '+token},
      payload:{message:message}
    });
    return{ok:res.getResponseCode()===200};
  }catch(err){
    return{ok:false,error:err.message};
  }
}
